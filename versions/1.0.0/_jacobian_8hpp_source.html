413 inline bool Jacobian::is_compatible(const State& state) const {
414  bool compatible = false;
415  if (state.get_type() == StateType::JOINTSTATE) {
416  // compatibility is assured through the vector of joint names
417  compatible = (this->get_name() == state.get_name())
418  && (this->cols_ == dynamic_cast<const JointState&>(state).get_size());
419  if (compatible) {
420  for (unsigned int i = 0; i < this->cols_; ++i) {
421  compatible = (compatible && this->joint_names_[i] == dynamic_cast<const JointState&>(state).get_names()[i]);
422  }
423  }
424  } else if (state.get_type() == StateType::CARTESIANSTATE) {
425  // compatibility is assured through the reference frame and the name of the frame
426  compatible = (this->reference_frame_ == dynamic_cast<const CartesianState&>(state).get_reference_frame())
427  && (this->frame_ == dynamic_cast<const CartesianState&>(state).get_name());
428  }
429  return compatible;
430 }
