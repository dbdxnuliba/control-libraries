583  return this->get_position();

592  return this->get_linear_velocity();
593 
594  case CartesianStateVariable::ANGULAR_VELOCITY:
595  return this->get_angular_velocity();
596 
597  case CartesianStateVariable::TWIST:
598  return this->get_twist();
599 
600  case CartesianStateVariable::LINEAR_ACCELERATION:
601  return this->get_linear_acceleration();
602 
603  case CartesianStateVariable::ANGULAR_ACCELERATION:
604  return this->get_angular_acceleration();
605 
606  case CartesianStateVariable::ACCELERATIONS:
607  return this->get_accelerations();
608 
609  case CartesianStateVariable::FORCE:
610  return this->get_force();
611 
612  case CartesianStateVariable::TORQUE:
613  return this->get_torque();
614 
615  case CartesianStateVariable::WRENCH:
616  return this->get_wrench();
617 
618  case CartesianStateVariable::ALL:
619  Eigen::VectorXd all_fields(25);

621  return all_fields;

624  return Eigen::Vector3d::Zero();

732 inline void CartesianState::set_state_variable(const Eigen::VectorXd& new_value,

743  case CartesianStateVariable::POSE:

745  break;
746 
747  case CartesianStateVariable::LINEAR_VELOCITY:
748  this->set_linear_velocity(new_value);
749  break;
750 
751  case CartesianStateVariable::ANGULAR_VELOCITY:
752  this->set_angular_velocity(new_value);
753  break;
754 
755  case CartesianStateVariable::TWIST:
756  this->set_twist(new_value);
757  break;
758 
759  case CartesianStateVariable::LINEAR_ACCELERATION:
760  this->set_linear_acceleration(new_value);
761  break;
762 
763  case CartesianStateVariable::ANGULAR_ACCELERATION:
764  this->set_angular_acceleration(new_value);
765  break;
766 
767  case CartesianStateVariable::ACCELERATIONS:
768  this->set_accelerations(new_value);
769  break;
770 
771  case CartesianStateVariable::FORCE:
772  this->set_force(new_value);
773  break;
774 
775  case CartesianStateVariable::TORQUE:
776  this->set_torque(new_value);
777  break;
778 
779  case CartesianStateVariable::WRENCH:
