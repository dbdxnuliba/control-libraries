123 inline void Linear<state_representation::CartesianState>::set_attractor(const state_representation::CartesianState& attractor) {
124  // validate that the reference frame of the attractor is always compatible with the DS reference frame
125  if (attractor.get_reference_frame() != this->get_base_frame().get_name()) {
126  if (attractor.get_reference_frame() != this->get_base_frame().get_reference_frame()) {
127  throw state_representation::exceptions::IncompatibleReferenceFramesException(
128  "The reference frame of the attractor " + attractor.get_name() + " in frame " + attractor.get_reference_frame()
129  + " is incompatible with the base frame of the dynamical system "
130  + this->get_base_frame().get_name() + " in frame " + this->get_base_frame().get_reference_frame() + "."
131  );
132  }
133  this->attractor_->set_value(this->get_base_frame().inverse() * attractor);
134  } else {
135  this->attractor_->set_value(attractor);
136  }
137 }
